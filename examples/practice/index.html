    # point pattern generator
def PointsGen(Xls, Yls):
    IDls = []
    for i in range(0, len(Xls)):
        IDls.append(i + 1)
    df_dict = {'ID': IDls, 'XCoord': Xls, 'YCoord': Yls}
    df = pd.DataFrame(df_dict)
    df = gpd.GeoDataFrame(df, geometry = gpd.points_from_xy(Xls, Yls))
    return df
